Using a for loop and if condition


function showEvenNumbers1(arr) {
  let result = [];
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] % 2 === 0) {
      result.push(arr[i]);
    }
  }
  return result;
}

Using the filter() method


function showEvenNumbers2(arr) {
  return arr.filter(function(number) {
    return number % 2 === 0;
  });
}

Using a Quick Sort


function quickSort(array) {
  if (array.length <= 1) {
    return array;
  }
  const pivot = array[array.length - 1];
  const left = [];
  const right = [];

  for (let i = 0; i < array.length - 1; i++) {
    if (array[i] < pivot) {
      left.push(array[i]);
    } else {
      right.push(array[i]);
    }
  }
  return [...quickSort(left), 
  pivot, ...quickSort(right)];
}
